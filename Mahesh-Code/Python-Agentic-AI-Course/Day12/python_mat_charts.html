# Bar chart using matplotlib
import matplotlib.pyplot as plt
batter = ['Rahul', 'Virat', 'Rohit', 'Shikhar', 'KL']
runs = [5000, 12000, 9000, 7000, 8000]

#create bar charts
plt.barh(batter, runs, color='red')
plt.xlabel('Batter')
plt.ylabel('Runs')
plt.title('Runs scored by Indian batsmen')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# Stock Price Data

import yfinance as yf
import matplotlib.pyplot as plt
import matplotlib as mpl

# Reset matplotlib configuration
#mpl.rcdefaults()

infy = yf.download('INFY.NS', start='2026-05-01', end='2026-05-16')

# Create line chart instead to show volume data
fig, ax = plt.subplots(figsize=(15, 5))
ax.plot(infy.index, infy['Volume'], marker='o', color='orange', linewidth=5, markersize=10)
ax.set_title('INFY Stock Volume')
ax.set_xlabel('Date')
ax.set_ylabel('Volume')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# Create Bar Chart Average Spending Score by Gender
import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv('Mall_Customers.csv')
avg_spending_score = df.groupby('Gender')["Spending Score (1-100)"].mean()
plt.bar(avg_spending_score.index, avg_spending_score.values, color=['blue', 'pink'])
plt.title('Average Spending Score by Gender')
plt.xlabel('Gender')
plt.ylabel('Average Spending Score')
plt.show()
